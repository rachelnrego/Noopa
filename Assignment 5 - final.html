import numpy as np
import matplotlib.pyplot as plt

## creating a function to visualize basis average,max and min
def visualize(filename):

    data = numpy.loadtxt(fname=filename, delimiter=',')

    fig = matplotlib.pyplot.figure(figsize=(10.0, 3.0))

    axes1 = fig.add_subplot(1, 3, 1)
     

   #axes1.set_ylabel('average')
    axes1.set_ylabel('Values')
    axes1.set_xlabel('Feature#')
    axes1.plot(numpy.mean(data, axis=0),label="Average")
    axes1.plot(numpy.max(data, axis=0),label="Max")
    axes1.plot(numpy.min(data, axis=0),label="Min")
      
    plt.legend(loc="upper left")
    fig.tight_layout()
    matplotlib.pyplot.show()
